# Windows Task Scheduler Configuration for Organize Files
#
# To set up the scheduled task on Windows:
#
# 1. Open Task Scheduler (taskschd.msc)
# 2. Click "Create Basic Task..."
# 3. Configure as follows:
#
# Name: Navidrome Organize Files
# Description: Automatically organize music files every hour
# 
# Trigger: Daily
# Start: Today
# Recur every: 1 days
# 
# Action: Start a program
# Program: C:\path\to\navidrome-cleanup\cron_organize_files.bat
# Start in: C:\path\to\navidrome-cleanup
#
# Settings:
# - Allow task to be run on demand: ✓
# - Run task as soon as possible after a scheduled start is missed: ✓
# - If the running task does not end when requested, force it to stop: ✓
# - Delete the task if it is not scheduled to run again: ✗
# - Run task only if computer is on AC power: ✗
# - Wake the computer to run this task: ✗
#
# Advanced Schedule Settings:
# - Repeat task every: 1 hour
# - for a duration of: 1 day
# - Expire: Never
#
# Alternative PowerShell command to create the task:
# 
# $action = New-ScheduledTaskAction -Execute "C:\path\to\navidrome-cleanup\cron_organize_files.bat" -WorkingDirectory "C:\path\to\navidrome-cleanup"
# $trigger = New-ScheduledTaskTrigger -Once -At (Get-Date).Date -RepetitionInterval (New-TimeSpan -Hours 1) -RepetitionDuration (New-TimeSpan -Days 1)
# $settings = New-ScheduledTaskSettingsSet -AllowStartIfOnBatteries -DontStopIfGoingOnBatteries -StartWhenAvailable
# Register-ScheduledTask -TaskName "Navidrome Organize Files" -Action $action -Trigger $trigger -Settings $settings -Description "Automatically organize music files every hour"
